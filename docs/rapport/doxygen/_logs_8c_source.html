116void logClientAction(int logsQueueId, Philosopher philosopher) {
117 const char *action;
118
119 // "Le philophe X est en train de Y : Z"
120 switch (philosopher.state) {
121 case THINKING:
122 action = "\x1B[1;34mpenser\x1B[0m :"; // mise en gras de l'action dans la même couleur
123 break;
124
125 case EATING:
126 action = "\x1B[1;32mmanger\x1B[0m :"; // mise en gras de l'action dans la même couleur
127 break;
128
129 default:
130 break;
131 }
132
133 char *message = (char *) malloc(512);
134
135 snprintf(
136 message,
137 512,
138 "%s %d %s %s %d%s",
139 "Le philosophe",
140 philosopher.id,
141 "est en train de",
142 action,
143 philosopher.stateTimer,
144 " secondes.\n"
145 );
146
147 logClientInfo(logsQueueId, message);
148 free(message);
149}
