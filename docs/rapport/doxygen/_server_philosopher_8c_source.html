74void definePhilosopherRightChopstick(ServerPhilosopher *philosopher, SharedResources *sharedResources) {
75 int lastPhilosopherId = sharedResources->numberPhilosophers;
76
77 Chopstick *firstChopstick = &sharedResources->chopsticks[0];
78 logServerState(sharedResources->logsQueueId, "Assignation de la baguette 1 (adresse %p) à droite du philosophe %d...\n", firstChopstick , philosopher->base.id);
79 philosopher->rightChopstick = firstChopstick;
80
81 // Attribution de la nouvelle baguette à droite de l'avant dernier philosophe, on vérifiant l'accès de son ancienne baguette pour éviter un déplacement en mémoire pendant l'utilisation
82 logServerState(sharedResources->logsQueueId, "Assignation de la baguette %d (adresse %p) à droite de l'avant denier philosophe %d...\n", philosopher->leftChopstick->id, philosopher->leftChopstick, sharedResources->philosophers[lastPhilosopherId - 1].base.id);
83
84 // Quand c'est le deuxième philosophe créé, le premier n'a pas de baguette à droite donc pas de sémaphore a tester
85 if (philosopher->base.id == 2) {
86 sharedResources->philosophers[0].rightChopstick = philosopher->leftChopstick;
87 }
88
89 else if (philosopher->base.id > 2) {
90 Chopstick *previousPhlosopherOldRightChopstick = sharedResources->philosophers[lastPhilosopherId - 1].rightChopstick;
91 sem_wait(&previousPhlosopherOldRightChopstick->usage);
92
93
94 sharedResources->philosophers[lastPhilosopherId - 1].rightChopstick = philosopher->leftChopstick;
95
96 sem_post(&previousPhlosopherOldRightChopstick->usage);
97 }
98}

112ServerPhilosopher createPhilosopher(SharedResources *sharedResources) {
113 sem_wait(&sharedResources->philosopherCreationProcess);
114
115 // Création d'un philosophe
116 int lastPhilosopherId = sharedResources->numberPhilosophers;
117
118 ServerPhilosopher philosopher;
119 memset(&philosopher, 0, sizeof(philosopher));
120
121 philosopher.base.id = lastPhilosopherId + 1;
122
123 logServerState(sharedResources->logsQueueId, "Création du philosophe %d...\n", philosopher.base.id);
124
125 // Création et Attribution de la baguette à sa gauche
126 philosopher.leftChopstick = createChopstick(philosopher.base.id, sharedResources);;
127
128 if (lastPhilosopherId > 0) {
129 definePhilosopherRightChopstick(&philosopher, sharedResources);
130 }
131
132 // Ajout dans la mémoire partagée
133 sharedResources->philosophers[lastPhilosopherId] = philosopher;
134 sharedResources->numberPhilosophers += 1;
135
136 // Incrémentation du nombre de philosophes qui peuvent manger en même
137 // Uniquement si le nouveau nombre de philosophes est un multiple de 2 (un philosophe sur deux peut manger)
138 if (sharedResources->numberPhilosophers % 2 == 0) {
139 sem_post(&sharedResources->maxAllowedEating);
140 }
141
142 sem_post(&sharedResources->philosopherCreationProcess);
143
144 return philosopher;
145}

161ServerPhilosopher *updatePhilosopher(Philosopher philosopher, SharedResources *sharedResources) {
162 ServerPhilosopher *serverPhilosopher = getPhilosopherFromId(philosopher.id, sharedResources->philosophers);
163
164 if (!serverPhilosopher) {
165 logClientInfo(sharedResources->logsQueueId, "Erreur, le philosophe à mettre à jour est introuvable dans la mémoire partagée.\n");
166 return NULL;
167 }
168
169 if (philosopher.state == THINKING) {
170
171 // S'il est passé de EATING a THINKING, libération des baguettes et incrémentation du compteur principal
172 if (serverPhilosopher->base.state == EATING) {
173
174 sem_post(&serverPhilosopher->leftChopstick->usage);
175 logClientInfo(sharedResources->logsQueueId, "Baguette gauche libérée\n");
176 logServerState(sharedResources->logsQueueId, "Le philosophe %d libère la baguette %d à sa gauche\n", philosopher.id, serverPhilosopher->leftChopstick->id);
177
178 sem_post(&serverPhilosopher->rightChopstick->usage);
179 logClientInfo(sharedResources->logsQueueId, "Baguette droite libérée\n");
180 logServerState(sharedResources->logsQueueId, "Le philosophe %d libère la baguette %d à sa droite\n", philosopher.id, serverPhilosopher->rightChopstick->id);
181
182
183 sem_post(&sharedResources->maxAllowedEating);
184 logClientInfo(sharedResources->logsQueueId, "Compteur libéré\n");
185 logServerState(sharedResources->logsQueueId, "Le philosophe %d libère le compteur\n\n", philosopher.id);
186
187 }
188
189 serverPhilosopher->base = philosopher;
190 logClientAction(sharedResources->logsQueueId, serverPhilosopher->base);
191 return NULL;
192 }
193
194 if (philosopher.state == EATING) {
195 serverPhilosopher->base = philosopher;
196 logClientAction(sharedResources->logsQueueId, serverPhilosopher->base);
197 return NULL;
198 }
199
200
201 if (philosopher.state == HUNGRY) {
202
203 serverPhilosopher->base = philosopher;
204
205 // On vérifie le compteur principal
206 // Si ça ne passe pas on log que le philosophe patiente et il est mis en attente
207
208 if (sem_trywait(&sharedResources->maxAllowedEating) == -1 && (errno == EAGAIN)) {
209 logServerState(sharedResources->logsQueueId, "Le philosophe %d attend que le compteur se libère\n", philosopher.id);
210 logClientInfo(sharedResources->logsQueueId, "En attente de pouvoir manger... \n");
211 sem_wait(&sharedResources->maxAllowedEating);
212 }
213
214 int allowedEating;
215 sem_getvalue(&sharedResources->maxAllowedEating, &allowedEating);
216 logServerState(sharedResources->logsQueueId, "Le philosophe %d s'ajoute au compteur (dispo restante : %d)\n", philosopher.id, allowedEating);
217
218 // Une fois le premier sémaphore pris, on vérifie les deux baguettes
219
220 // On vérifie une première baguette (non bloquant)
221 // Si ça ne passe on log que le philosophe patiente et il est mis en attente
222 if (sem_trywait(&serverPhilosopher->leftChopstick->usage) == -1 && (errno == EAGAIN)) {
223 logServerState(sharedResources->logsQueueId, "Le philosophe %d attend que la baguette %d à sa gauche se libère\n", philosopher.id, serverPhilosopher->leftChopstick->id);
224 logClientInfo(sharedResources->logsQueueId, "En attente de la baguette gauche...\n");
225 sem_wait(&serverPhilosopher->leftChopstick->usage);
226 }
227
228
229 logServerState(sharedResources->logsQueueId, "Le philosophe %d prend la baguette %d à sa gauche\n", philosopher.id, serverPhilosopher->leftChopstick->id);
230
231
232 // On vérifie la seconde baguette (non bloquant)
233 // Si ça ne passe on log que le philosophe patiente et il est mis en attente
234 if (sem_trywait(&serverPhilosopher->rightChopstick->usage) == -1 && (errno == EAGAIN)) {
235 logServerState(sharedResources->logsQueueId, "Le philosophe %d attend que la baguette %d à sa droite se libère\n", philosopher.id, serverPhilosopher->rightChopstick->id);
236 logClientInfo(sharedResources->logsQueueId, "En attente de la baguette droite...\n");
237 sem_wait(&serverPhilosopher->rightChopstick->usage);
238 }
239
240 logServerState(sharedResources->logsQueueId, "Le philosophe %d prend la baguette %d à sa droite\n", philosopher.id, serverPhilosopher->rightChopstick->id);
241
242 // Ensuite on peut passer à l'état EATING et envoyer la réponse au client qui attend une réponse
243 serverPhilosopher->base.state = EATING;
244 serverPhilosopher->base.stateTimer = 0;
245
246 return serverPhilosopher;
247 }

249 return NULL;
